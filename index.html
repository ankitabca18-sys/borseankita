class Product: def __init__(self, name, price, quantity): self.name = name self.price = price self.quantity = quantity def total_price(self): return self.price * self.quantity def display_info(self): print(f"Product: {self.name}") print(f"Price: ${self.price:.2f}") print(f"Quantity: {self.quantity}") print(f"Total Price: ${self.total_price():.2f}") # Example usage product1 = Product("Laptop", 999.99, 2) product1.display_info()
